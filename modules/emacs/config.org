* ORG Roam

#+BEGIN_SRC emacs-lisp
(use-package org-roam
  :ensure t
  :custom
  (org-roam-directory (file-truename "~/Notes/"))
  (org-roam-node-display-template "${formatted-tags:30} ${title:*}")
  (org-roam-completion-everywhere t)
  :bind (("C-c n l" . org-roam-buffer-toggle)
         ("C-c n f" . org-roam-node-find)
         ("C-c n g" . org-roam-graph)
         ("C-c n i" . org-roam-node-insert)
         ("C-c n c" . org-roam-capture)
         ("C-c n j" . org-roam-dailies-capture-today)
         :map org-mode-map
         ("C-M-i" . completion-at-point))
  :config
  ;; Custom accessor to format tags with # prefix
  (cl-defmethod org-roam-node-formatted-tags ((node org-roam-node))
    "Format tags with # prefix for display"
    (let ((tags (org-roam-node-tags node)))
      (if tags
          (mapconcat (lambda (tag) (concat "#" tag)) tags " ")
        "")))

  (org-roam-db-autosync-mode))
#+END_SRC
