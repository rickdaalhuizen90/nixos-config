* Performance Configuration

#+BEGIN_SRC emacs-lisp
;; Increase GC threshold for better startup performance
(setq gc-cons-threshold (* 50 1000 1000))
(setq gc-cons-percentage 0.6)
(add-hook 'focus-out-hook #'garbage-collect)

;; Disable file watching for better performance
(setq make-backup-files nil
      auto-save-default nil
      create-lockfiles nil
      auto-revert-check-vc-info nil)
#+END_SRC

* UI Configuration

#+BEGIN_SRC emacs-lisp
(defvar runemacs/default-font-size 140)

;; Clean UI
(setq inhibit-startup-message t
      visible-bell t)

(dolist (mode '(scroll-bar-mode tool-bar-mode tooltip-mode menu-bar-mode))
  (when (fboundp mode) (funcall mode -1)))

(set-fringe-mode 10)
(column-number-mode)
(global-display-line-numbers-mode t)

;; Disable line numbers for text-focused modes
(dolist (mode '(org-mode-hook term-mode-hook shell-mode-hook eshell-mode-hook))
  (add-hook mode (lambda () (display-line-numbers-mode 0))))
#+END_SRC

* Evil Mode

#+BEGIN_SRC emacs-lisp
(use-package evil
  :init
  (setq evil-want-integration t
        evil-want-keybinding nil
        evil-want-C-u-scroll t)
  :config
  (evil-mode 1)
  (define-key evil-insert-state-map (kbd "C-g") 'evil-normal-state)

  ;; Visual line navigation
  (evil-global-set-key 'motion "j" 'evil-next-visual-line)
  (evil-global-set-key 'motion "k" 'evil-previous-visual-line))

(use-package evil-collection
  :after evil
  :config (evil-collection-init))

* Org Mode

#+BEGIN_SRC emacs-lisp
(use-package org
  :hook (org-mode . (lambda ()
                      (org-indent-mode 1)
                      (variable-pitch-mode 1)
                      (visual-line-mode 1)))
  :config
  (setq org-startup-indented t
        org-hide-emphasis-markers t
        org-ellipsis " â–¾"
        org-list-allow-alphabetical t
        org-display-inline-images t
        org-startup-with-inline-images t
        org-adapt-indentation nil

        ;; Enhanced folding/visibility
        org-startup-folded 'content
        org-cycle-separator-lines 2

        ;; Better TODO keywords
        org-todo-keywords '((sequence "TODO" "IN-PROGRESS" "|" "DONE" "CANCELLED"))
        org-todo-keyword-faces '(("IN-PROGRESS" . "orange"))

        ;; Archive settings
        org-archive-location "%s_archive::* Archived Tasks"

        ;; Search and navigation
        org-imenu-depth 3                    ; Better navigation with imenu
        org-goto-auto-isearch t              ; Auto search in goto

        ;; Links
        org-return-follows-link t            ; RET follows links
        org-link-descriptive t               ; Hide link URLs, show descriptions

        ;; Timestamps and clocking
        org-log-done 'time                   ; Log when tasks completed
        org-log-into-drawer t                ; Put logs in :LOGBOOK: drawer
        org-clock-idle-time 15               ; Resolve idle time after 15 min

        ;; Tables
        org-table-auto-blank-field t         ; Clear field when typing

        ;; Source blocks (for notes with code)
        org-src-fontify-natively t           ; Syntax highlight in src blocks
        org-src-tab-acts-natively t          ; Tab works normally in src blocks
        org-confirm-babel-evaluate nil)      ; Don't ask before running code

  ;; Quick access to files
  (setq org-agenda-files '("~/Notes/"))

  ;; Better refile targets
  (setq org-refile-targets '((org-agenda-files :maxlevel . 3))
        org-refile-use-outline-path 'file
        org-outline-path-complete-in-steps nil))
#+END_SRC
