  :ensure nil   ;; use built-in Org, avoid ELPA org 404
  :hook (org-mode . (lambda ()
                      (org-indent-mode 1)
                      (variable-pitch-mode 1)
                      (visual-line-mode 1)))
  :config
  (setq org-startup-indented t
        org-hide-emphasis-markers t
        org-ellipsis " â–¾"
        org-list-allow-alphabetical t
        org-display-inline-images t
        org-startup-with-inline-images t

        ;; Enhanced folding/visibility
        org-startup-folded 'content
        org-cycle-separator-lines 2

        org-todo-keywords '((sequence "TODO" "IN-PROGRESS" "|" "DONE" "CANCELLED"))

        org-archive-location "%s_archive::* Archived Tasks"

* Spell checking (Hunspell + Flyspell)

#+BEGIN_SRC emacs-lisp
;; Use hunspell as the spell checker
(setq ispell-program-name "hunspell")
(setq ispell-dictionary "en_US")

;; Optional: tell Hunspell where to look for dictionaries, if needed.
;; Usually not necessary on NixOS, but you can uncomment and adjust:
;; (setenv "DICPATH" (concat (getenv "HOME") "/.nix-profile/share/hunspell"))

;; Enable spell checking in writing modes
(dolist (hook '(text-mode-hook
                markdown-mode-hook
                org-mode-hook))
  (add-hook hook #'flyspell-mode))

;; In programming modes, only check comments/strings:
(add-hook 'prog-mode-hook #'flyspell-prog-mode)

;; Handy keybindings for corrections
(with-eval-after-load 'flyspell
  ;; Correct word at point
  (define-key flyspell-mode-map (kbd "C-;") #'flyspell-correct-word-before-point)
  ;; Move to next error
  (define-key flyspell-mode-map (kbd "M-]") #'flyspell-goto-next-error))
#+END_SRC

* Desktop Mode

#+BEGIN_SRC emacs-lisp
;; Store session files in ~/.emacs.d
(setq desktop-dirname (expand-file-name "~/.emacs.d/")
      desktop-path (list desktop-dirname)
      desktop-base-file-name "emacs.desktop"
      desktop-save 'if-exists          ;; save automatically if file exists
      desktop-load-locked-desktop t    ;; load even if marked "in use"
      desktop-restore-eager 10)        ;; restore first few buffers quickly

(desktop-save-mode 1)

;; Remember cursor position in files
(save-place-mode 1)
